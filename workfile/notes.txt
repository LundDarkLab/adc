id int (chiave primaria)             
name varchar (nome dell'oggeto)    
inventory varchar (numero di inventario)     
conservation_state varchar (stato di conservazione dell'oggetto: buono, mediocre ecc)     
object_condition varchar (indica se l'oggetto è da restaurare o no)     
is_museum_copy tinyint (campo booleano che indica se l'oggetto è originale o è una copia)      
category_class varchar (indica genericamente cos'è l'oggetto archeologico: ascia, vaso, monile ecc.)     
category_specs varchar (indica una specifica tipologia della classe del campo precedente)     
type text (campo libero che descrive brevemente particolari della categoria specifica)            
start int (indica l'anno in cui l'oggetto compare per la prima volta)             
end int (indica l'anno in cui orientativamente l'oggetto non è stato più utilizzato)            
description text (descrizione dell'oggetto)            
notes text (note sull'oggetto)            
author text (l'utente che ha creato il record)              
owner text (chi detiene i diritti di proprietà dell'oggetto) 
license text (tipo di licenza utilizzato) 
created_at timestamp (data di creazione)
